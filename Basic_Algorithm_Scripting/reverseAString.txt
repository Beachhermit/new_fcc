function reverseString(str) {

    //first I tried reading str backwards and writing it one by one into a new string

    //this didn't work because strings are immutable, duh


    
//loop that reads str backwards and pushes onto new array
    let backwards = [];


    for (let i = str.length-1; i >= 0; i--) {

        backwards.push(str[i]);

    }


    //now we have an array with the letters in correct order

    //need to convert array to string without the commas

    let backwardsString = backwards.join("");


  
  return backwardsString;

}
